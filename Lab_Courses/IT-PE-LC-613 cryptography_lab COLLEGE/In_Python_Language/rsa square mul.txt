import math
import sys

def gcd(a, b):
	while b:
		remainder = a % b
		a = b
		b = remainder
	return a
	
def square_mul(base, exp, mod):

	if exp == 0:
		return 1 % mod
	if exp == 1:
		return base % mod
		
	res = 1
	current_base = base % mod
	
	while exp > 0:
		#if exponent is odd 
		if not(exp % 2 == 0): 
			res = (res * current_base) % mod
						
		current_base = (current_base * current_base) % mod
		exp = exp // 2 
        	
	return res

def encrypt(public_key, plaintext):
	e, n = public_key
	cipher_text = []
	for char in plaintext:
		cascii = ord(char)
		encrypt_char = square_mul(cascii, e, n)
		cipher_text.append(encrypt_char)
	return cipher_text
	
def decrypt(private_key, ciphertext):
	d, n = private_key
	plaintext = []
	for char in ciphertext:
		cascii = square_mul(char, d, n)
		decrypt_char = chr(cascii)
		plaintext.append(decrypt_char)
	return "".join(plaintext)
