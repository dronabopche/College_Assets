%{
#include<stdio.h>
#include<ctype.h>
int yylval;
%}
%token id
%left "+" "-"
%left "*" "/"
%%
S: E '\n' { printf("%d", $1); }	
;

E: E '+' E { $$ = $1 + $3; }
 | E '*' E { $$ = $1 * $3; }
 | E '-' E { $$ = $1 - $3; }
 | E '/' E { $$ = $1 / $3; }
  | num     { $$ = $1; }
num:num id{ $$=10*$1+$2; }
 | id      { $$ = $1; }
;
%%

int main()
{
    yyparse();
    return 0;
}

int yylex()
{
    int c;
    while ((c = getchar()) == ' ' || c == '\t'); // skip whitespace

    if (isdigit(c))
    {
        yylval = 0;
        while (isdigit(c)) {
            yylval = yylval * 10 + (c - '0');
            c = getchar();
        }
        ungetc(c, stdin); // push back the non-digit character
        return id;
    }

    return c;
}

int yyerror()
{
    return 0;
}

int yywrap()
{
    return 0;
}
